  (function() {

    var deckElement = document.getElementById("raw-deck-name-back");

    if (deckElement && deckElement.textContent.indexOf("{{") === -1) {

      var fullDeck = deckElement.textContent.trim();

      var deckParts = fullDeck.split("::");

      var subDeck = deckParts.pop();

      var parentDecks = deckParts.length > 0 ? deckParts.join(" • ") : "MAIN DECK";


      var parentEls = document.querySelectorAll(".parent-deck-text");

      var subEls = document.querySelectorAll(".sub-deck-text");


      parentEls.forEach(function(el) { el.textContent = parentDecks.toUpperCase(); });

      subEls.forEach(function(el) { el.textContent = subDeck.toUpperCase(); });

    }

  })();


  function openModal(imgHtml) {

    var overlay = document.createElement('div');

    overlay.className = 'anki-modal-overlay';

    overlay.onclick = function() { document.body.removeChild(overlay); };


    var content = document.createElement('div');

    content.className = 'anki-modal-content';

    content.innerHTML = imgHtml;


    var closeBtn = document.createElement('div');

    closeBtn.className = 'anki-modal-close';

    closeBtn.innerHTML = '×';


    overlay.appendChild(content);

    overlay.appendChild(closeBtn);

    document.body.appendChild(overlay);

  }

</script>
